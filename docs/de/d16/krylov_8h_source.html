150 template<class Real>
151 void GeneratePlaneRotation(const Real dx, const Real dy, Real &cs, Real &sn) {
152  if (dy == 0.0) {
153  cs = 1.0;
154  sn = 0.0;
155  }
156  else if (std::abs(dy) > std::abs(dx)) {
157  Real temp = dx / dy;
158  sn = 1.0 / sqrt(1.0 + temp * temp);
159  cs = temp * sn;
160  }
161  else {
162  Real temp = dy / dx;
163  cs = 1.0 / sqrt(1.0 + temp * temp);
164  sn = temp * cs;
165  }

176 void Update(Vector &x, int k, const Matrix &h, const Vector &s,
177  const std::vector<Vector> &v, const std::vector<Vector> &z,
178  const Preconditioner &R, bool is_fgmres) {
179  Vector y(s);
180 
181  // Backsolve: y(m) = argmin || beta*e(1) - H(m)*y ||
182  for (int i = k; i >= 0; i--) {
183  y(i) = (h(i, i) != 0.0) ? (y(i) / h(i, i)) : 0.0;
184  for (int j = i - 1; j >= 0; j--)
185  y(j) -= h(j, i) * y(i);
186  }
187 
188  // Update the solution vector;
189  if (R.isIdentity()) {
190  // x(m) = x(0) + V(m)*y(m)
191  for (int j = 0; j <= k; j++)
192  x += v[j] * y(j);
193  }
194  else if (is_fgmres) {
195  // x(m) = x(0) + Z(m)*y(m) since Z = R*V
196  for (int j = 0; j <= k; j++)
197  x += z[j] * y(j);
198  }
199  else {
200  // x(m) = x(0) + R * (V(m)*y(m))
201  Vector xx(x.size());
202  std::fill(xx.begin(), xx.end(), 0.0);
203  for (int j = 0; j <= k; j++)
204  xx += v[j] * y(j);
205  x += R * xx;
206  }

333 template < class ResidualFunctor, class Vector, class VectorB, class Real >
334 int
335 BiCGSTAB(const ResidualFunctor &A, Vector &x, const VectorB &b,
336  const Preconditioner &M, size_t &max_iter, Real &tol,
337  NullIterPrintOut &printOut) {
338  Real resid;
339  Real rho_1, rho_2 = 0, alpha = 0, beta = 0, omega = 0;
340  Vector p, phat, s, shat, t, v;
341 
342  Real normb = norm_2(b);
343  Vector r = b - A.product(x);
344  Vector rtilde = r;
345 
346  if (less_than_machine_epsilon(normb * tol))
347  normb = 1;
348 
349  if ((resid = norm_2(r) / normb) <= tol) {
350  tol = resid;
351  max_iter = 0;
352  printOut.onConvergence(AN_BICGSTAB, max_iter, tol);
353  return 0;
354  }
355 
356  for (size_t i = 1; i <= max_iter; i++) {
357  rho_1 = inner_prod(rtilde, r);
358  if (rho_1 == 0) {
359  tol = norm_2(r) / normb;
360  printOut.onFailure(AN_BICGSTAB, i, CFR_BREAKDOWN);
361  return 2;
362  }
363  if (i == 1)
364  p = r;
365  else {
366  beta = (rho_1 / rho_2) * (alpha / omega);
367  p = r + beta * (p - omega * v);
368  }
369  phat = M * p;
370  v = A.product(phat);
371  alpha = rho_1 / inner_prod(rtilde, v);
372  s = r - alpha * v;
373  resid = norm_2(s) / normb;
374  printOut.onIteration(AN_BICGSTAB, i, resid);
375  if (resid < tol) {
376  x += alpha * phat;
377  max_iter = i;
378  tol = resid;
379  printOut.onConvergence(AN_BICGSTAB, max_iter, tol);
380  return 0;
381  }
382  shat = M * s;
383  t = A.product(shat);
384  omega = inner_prod(t, s) / inner_prod(t, t);
385  x += alpha * phat + omega * shat;
386  r = s - omega * t;
387 
388  rho_2 = rho_1;
389  resid = norm_2(r) / normb;
390  printOut.onIteration(AN_BICGSTAB, i, resid);
391  if (resid < tol) {
392  tol = resid;
393  max_iter = i;
394  printOut.onConvergence(AN_BICGSTAB, max_iter, tol);
395  return 0;
396  }
397  if (omega == 0 || std::isnan((Real)resid)) {
398  tol = norm_2(r) / normb;
399  printOut.onFailure(AN_BICGSTAB, i, CFR_BREAKDOWN);
400  return 3;
401  }
402  }
403 
404  tol = resid;
405  printOut.onFailure(AN_BICGSTAB, max_iter, CFR_TOO_MANY_ITERATIONS);
406  return 1;
407 }
